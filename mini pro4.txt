#include <stdio.h>
#include <string.h>

int main() 
{
    char sentence[200];
    printf("Enter a sentence: ");
    fgets(sentence, sizeof(sentence), stdin);
    sentence[strcspn(sentence, "\n")] = '\0';

    printf("Words in the sentence:\n");
    char *word = strtok(sentence, " ");
    while (word != NULL)
 {
        printf("%s\n", word); 
        word = strtok(NULL, " ");
    }

    return 0;
}
